    How to use

    # Install

    pnpm add -D vuepress-plugin-comment2

    yarn add -D vuepress-plugin-comment2

    npm i -D vuepress-plugin-comment2

    # Usage

    // .vuepress/config.ts
import { commentPlugin } from "vuepress-plugin-comment2";

export default {
  plugins: [
    commentPlugin({
      // your options
    }),
  ],
};

    // .vuepress/config.js
import { commentPlugin } from "vuepress-plugin-comment2";

export default {
  plugins: [
    commentPlugin({
      // your options
    }),
  ],
};
